11  /**
12   * Examples taken from Maven EAR Plugin
13   *
14   * Invoker Integration Test:
15   * <ul>
16   *   <li>test_project_root</li>
17   *   <li>packaging_includes</li>
18   *   <li>packaging_excludes</li>
19   *   <li>transitive_excludes</li>
20   * </ul>
21   *
22   * @author Karl Heinz Marbaise
23   */
24  @MavenIT
25  class EARIT {
26  
27    @MavenTest
28    void basic(MavenExecutionResult result, MavenProjectResult project) {
29      assertThat(result).isSuccessful();
30      assertThat(project).hasTarget()
31          .withEarFile()
32          .containsOnlyOnce("META-INF/application.xml", "META-INF/appserver-application.xml");
33    }
34  
35    @MavenTest
36    void packaging_includes(MavenExecutionResult result, MavenProjectResult project) {
37      assertThat(result).isSuccessful();
38      assertThat(project).hasTarget()
39          .withEarFile()
40          .doesNotContain("commons-io-commons-io-1.4.jar")
41          .containsOnlyOnce("commons-lang-commons-lang-2.5.jar", "META-INF/application.xml", "META-INF/MANIFEST.MF");
42    }
43  
44    @MavenTest
45    void packaging_excludes(MavenExecutionResult result, MavenProjectResult project) {
46      assertThat(result).isSuccessful();
47      assertThat(project).hasTarget()
48          .withEarFile()
49          .doesNotContain("commons-lang-commons-lang-2.5.jar")
50          .containsOnlyOnce("META-INF/application.xml", "META-INF/MANIFEST.MF");
51    }
52  
53    @MavenTest
54    void resource_custom_directory(MavenExecutionResult result, MavenProjectResult project, MavenLog log) {
55      assertThat(result).isSuccessful();
56      assertThat(log).isSuccessful();
57      assertThat(project).hasTarget()
58          .withEarFile()
59          .containsOnlyOnce("META-INF/application.xml", "APP-INF/classes/foo.properties");
60    }
61  
62    @MavenTest
63    void transitive_excludes(MavenExecutionResult result, MavenProjectResult project, MavenLog log) {
64      assertThat(result).isSuccessful();
65      assertThat(log).isSuccessful();
66      assertThat(log).info().stream().filter(s -> s.startsWith("TESTING"));
67      assertThat(log).debug().stream().filter(s -> s.endsWith("FAILURE"));
68      //assertThat(log).buildFailure(); DOES NOT WORK YET
69  
70      assertThat(project).hasTarget()
71          .withEarFile()
72          .containsOnlyOnce("org.apache.maven-maven-core-3.0.jar", "META-INF/application.xml");
73    }
74  
75    @MavenTest
76    void skinny_wars_javaee5(MavenExecutionResult result, MavenProjectResult project, MavenLog log) {
77      assertThat(result).isSuccessful();
78      assertThat(project)
79          .hasModule("war-module")
80          .hasModule("ear-module");
81  
82      /*
83      target (issue-4-maven-log-file-assertions *)$ unzip -t ear-module-1.0.ear
84  Archive:  ear-module-1.0.ear
85      testing: META-INF/MANIFEST.MF     OK
86      testing: META-INF/                OK
87      testing: lib/                     OK
88      testing: META-INF/maven/          OK
89      testing: META-INF/maven/org.apache.maven.its.ear.skinnywars/   OK
90      testing: META-INF/maven/org.apache.maven.its.ear.skinnywars/ear-module/   OK
91      testing: org.apache.maven.its.ear.skinnywars-war-module-1.0.war   OK
92      testing: META-INF/application.xml   OK

102             "lib/commons-lang-commons-lang-2.5.jar",
103             "META-INF/application.xml");
104   }
