90    @Override
91    public boolean supportsParameter(ParameterContext parameterContext, ExtensionContext extensionContext)
92        throws ParameterResolutionException {
93      return Stream.of(ParameterType.values())
94          .anyMatch(parameterType -> parameterType.getKlass() == parameterContext.getParameter().getType());
95    }
