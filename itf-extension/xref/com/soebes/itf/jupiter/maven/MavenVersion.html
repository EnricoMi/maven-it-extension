22  import static org.apiguardian.api.API.Status.EXPERIMENTAL;
23  
24  import java.util.stream.Stream;
25  import org.apiguardian.api.API;
26  import org.junit.platform.commons.logging.Logger;
27  import org.junit.platform.commons.logging.LoggerFactory;
28  import org.junit.platform.commons.util.StringUtils;
29  
30  /**
31   * @author Karl Heinz Marbaise
32   */
33  @API(status = EXPERIMENTAL, since = "0.1.0")
34  public enum MavenVersion {
35    UNKNOWN("UNKOWN"), //TODO: Need to reconsider?
36    M3_0_5("3.0.5"),
37    M3_1_1("3.1.1"),
38    M3_2_5("3.2.5"),
39    M3_3_1("3.3.1"),
40    M3_3_9("3.3.9"),
41    M3_5_0("3.5.0"),
42    M3_5_2("3.5.2"),
43    M3_5_3("3.5.3"),
44    M3_5_4("3.5.4"),
45    M3_6_0("3.6.0"),
46    M3_6_1("3.6.1"),
47    M3_6_2("3.6.2"),
48    M3_6_3("3.6.3");
49  
50    private static final Logger logger = LoggerFactory.getLogger(MavenVersion.class);
51  
52    private static final MavenVersion CURRENT_MAVEN_VERSION = determineCurrentVersion();
53  
54    private final String versionString;
55  
56    MavenVersion(String versionString) {
57      this.versionString = versionString;
58    }
59  
60    private static MavenVersion determineCurrentVersion() {
61      String currentVersion = System.getProperty("maven.version");
62      //FIXME: Think about usage of interal API?
63      if (StringUtils.isBlank(currentVersion)) {
64        logger.error(
65            () -> "JVM system property 'maven.version' is undefined. It is therefore not possible to detect Maven version.");
66      }
67      return Stream.of(values())
68          .filter(mavenVersion -> mavenVersion.getVersionString().equals(currentVersion))
69          .findFirst()
70          .orElseGet(() -> MavenVersion.UNKNOWN);
71    }
72  
73    /**
74     * @return {@code true} if <em>this</em> {@link MavenVersion} is known to be the Maven version for the currently being
75     * executed Maven version.
76     */
77    public boolean isCurrentVersion() {
78      return this == CURRENT_MAVEN_VERSION;
79    }
80  
81    public String getVersionString() {
82      return versionString;
83    }
84  }
