22  import static org.apiguardian.api.API.Status.EXPERIMENTAL;
23  
24  import java.util.StringJoiner;
25  import org.apiguardian.api.API;
26  
27  /**
28   * @author Karl Heinz Marbaise
29   */
30  @API(status = EXPERIMENTAL, since = "0.1.0")
31  public class MavenExecutionResult {
32  
33    private final ExecutionResult result;
34  
35    private final int returnCode;
36  
37    private final MavenLog mavenLog;
38  
39    private final MavenProjectResult mavenProjectResult;
40  
41    private final MavenCacheResult mavenCacheResult;
42  
43    public MavenExecutionResult(ExecutionResult result, int returnCode, MavenLog mavenLog,
44        MavenProjectResult mavenProjectResult, MavenCacheResult mavenCacheResult) {
45      this.result = result;
46      this.returnCode = returnCode;
47      this.mavenLog = mavenLog;
48      this.mavenProjectResult = mavenProjectResult;
49      this.mavenCacheResult = mavenCacheResult;
50    }
51  
52    public int getReturnCode() {
53      return returnCode;
54    }
55  
56    public boolean isSuccesful() {
57      return ExecutionResult.Successful.equals(this.result);
58    }
59  
60    public boolean isFailure() {
61      return ExecutionResult.Failure.equals(this.result);
62    }
63  
64    public boolean isError() {
65      return ExecutionResult.Error.equals(this.result);
66    }
67  
68    public ExecutionResult getResult() {
69      return this.result;
70    }
71  
72    public MavenLog getMavenLog() {
73      return mavenLog;
74    }
75  
76    public MavenCacheResult getMavenCacheResult() {
77      return mavenCacheResult;
78    }
79  
80    public MavenProjectResult getMavenProjectResult() {
81      return mavenProjectResult;
82    }
83  
84    public enum ExecutionResult {
85      Successful,
86      Failure,
87      Error
88    }
89  
90    @Override
91    public String toString() {
92      return new StringJoiner(", ", MavenExecutionResult.class.getSimpleName() + "[", "]").add("result=" + result)
93          .add("returnCode=" + returnCode)
94          .add("mavenLog=" + mavenLog)
95          .add("mavenProjectResult=" + mavenProjectResult)
96          .add("mavenCacheResult=" + mavenCacheResult)
97          .toString();
98    }
99  }
