36  final class StorageHelper {
37  
38    private static final Namespace NAMESPACE_MAVEN_IT = Namespace.create(MavenITExtension.class);
39  
40    private final Store nameSpace;
41  
42    private final ExtensionContext context;
43  
44    StorageHelper(ExtensionContext context) {
45      this.nameSpace = context.getStore(NAMESPACE_MAVEN_IT);
46      this.context = context;
47    }
48  
49    <V> V get(Storage storage, Class<V> requiredType) {
50      return nameSpace.get(storage, requiredType);
51    }
52  
53    <V> V get(Object key, Class<V> requiredType) {
54      return nameSpace.get(key, requiredType);
55    }
56  
57    void put(Object key, Object value) {
58      nameSpace.put(key, value);
59    }
60  
61    void save(File mavenItBaseDirectory, File mavenItTestCaseBaseDirectory, File targetDirectory) {
62      StorageHelpersion/StorageHelper.html#StorageHelper">StorageHelper sh = new StorageHelper(context);
63      sh.put(Storage.MAVEN_IT_BASEDIRECTORY, mavenItBaseDirectory);
64      sh.put(Storage.MAVEN_IT_TESTCASE_BASEDIRECTORY, mavenItTestCaseBaseDirectory);
65      sh.put(Storage.TARGET_DIRECTORY, targetDirectory);
66    }
67  
68    void save(MavenExecutionResult result, MavenLog log, MavenCacheResult mavenCacheResult,
69        MavenProjectResult mavenProjectResult) {
70      put(ParameterType.ExecutionResult + context.getUniqueId(), result);
71      put(ParameterType.LogResult + context.getUniqueId(), log);
72      put(ParameterType.CacheResult + context.getUniqueId(), mavenCacheResult);
73      put(ParameterType.ProjectResult + context.getUniqueId(), mavenProjectResult);
74    }
75  }
