26  	/**
27  	 * The test will check that the jar of the test project will be installed into the
28  	 * {@code invoker-repo} and the appropriate {@codoe pom.xml} file.
29  	 *
30  	 * The given test project does not have any dependencies nor any code.
31  	 *
32  	 * We are using {@code pre-integration-test} cause we want to test only the
33  	 * plugin itself.
34  	 *
35  	 * @param result {@link MavenExecutionResult}
36  	 */
37  	@MavenTest(options = {MavenOptions.NO_TRANSFER_PROGRESS}, goals = {"pre-integration-test"})
38  	@DisplayName("Running within an execution block in pom file with given goal but no explicit life cycle binding.")
39  	void install_should_not_fail(MavenExecutionResult result) {
40  		MavenITAssertions.assertThat(result).isSuccessful().project();
41  
42  		//FIXME: The following should be made easier See https://github.com/khmarbaise/maven-it-extension/issues/39
43  		File target = new File(result.getMavenProjectResult().getBaseDir(), "target");
44  		File invokerRepo = new File(target, "invoker-repo");
45  
46  		File install_should_not_fail = new File(invokerRepo, "com/soebes/itf/maven/plugin/its/install_should_not_fail");
47  
48  		assertThat(install_should_not_fail).isDirectory().satisfies(isnf -> {
49  			assertThat(new File(isnf, "maven-metadata-local.xml")).isNotEmpty();
50  			assertThat(new File(isnf, "1.0")).isDirectory().satisfies(v10 -> {
51  				assertThat(v10).satisfies(file -> {
52  					assertThat(new File(file, "install_should_not_fail-1.0.jar")).isNotEmpty();
53  					assertThat(new File(file, "install_should_not_fail-1.0.pom")).isNotEmpty();
54  					assertThat(new File(file, "_remote.repositories")).isNotEmpty();
55  				});
56  			});
57  		});
58  
59  	}

106 	/**
107 	 * The test will check that the jar of the test project and the single dependency
108 	 * will be installed into the {@code invoker-repo} and the appropriate {@codoe pom.xml} file
109 	 * including their transitive dependencies.
110 	 *
111 	 * The given test project has a single dependency but no code at all.
112 	 * The single dependency is <a href="https://search.maven.org/artifact/junit/junit/4.13/jar">junit</a>
113 	 * which has a single dependency <a href="https://search.maven.org/artifact/org.hamcrest/hamcrest-core/1.3/jar">hamcrest-core</a>
114 	 * (scope:compile). Furthermore the hamcrest-core has a
115 	 * <a href="https://search.maven.org/artifact/org.hamcrest/hamcrest-parent/1.3/pom">hamcrest-parent</a>
116 	 * which needs to be installed as well.
117 	 *
118 	 * We are using {@code pre-integration-test} cause we want to test only the
119 	 * plugin itself.
120 	 *
121 	 * @param result {@link MavenExecutionResult}
122 	 */
123 	@MavenTest(options = {MavenOptions.NO_TRANSFER_PROGRESS}, goals = {"pre-integration-test"})
124 	@DisplayName("Install with a single dependency and one transitive dependency")
125 	void install_with_dep_and_transitive_dep(MavenExecutionResult result) {
126 		MavenITAssertions.assertThat(result).isSuccessful().project();
127 
128 		File target = new File(result.getMavenProjectResult().getBaseDir(), "target");
129 		File invokerRepo = new File(target, "invoker-repo");
130 
131 		File install_should_not_fail = new File(invokerRepo, "com/soebes/itf/maven/plugin/its/install_with_dep_and_transitive_dep");
132 		assertThat(install_should_not_fail).isDirectory().satisfies(isnf -> {
133 			assertThat(new File(isnf, "1.0")).isDirectory().satisfies(v10 -> {
134 				assertThat(v10).satisfies(file -> {
135 					assertThat(new File(file, "install_with_dep_and_transitive_dep-1.0.jar")).isNotEmpty();
136 					assertThat(new File(file, "install_with_dep_and_transitive_dep-1.0.pom")).isNotEmpty();
137 				});
138 			});
139 		});
140 
141 		File junit = new File(invokerRepo, "junit/junit");
142 		assertThat(junit).isDirectory().satisfies(isnf -> {
143 			assertThat(new File(isnf, "4.13")).isDirectory().satisfies(v4_13 -> {
